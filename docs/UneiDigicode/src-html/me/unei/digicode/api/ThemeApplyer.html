001package me.unei.digicode.api;
002
003import me.unei.digicode.mp.APlayer;
004import me.unei.digicode.mp.AnInventory;
005import me.unei.digicode.mp.AnItem;
006
007/**
008 * Represents a theme generator (inventory creation).
009 * 
010 * @author JÃ«sFot
011 */
012public interface ThemeApplyer
013{
014        /**
015         * Gets the of the inventory view to generate.
016         * <p>
017         * Recommended: 6*9
018         * 
019         * @return Returns the inventory size.
020         */
021        public int getInventorySize();
022        
023        /**
024         * Checks whenever a given {@linkplain InterfaceElements element} has to be shown to a player.
025         * 
026         * @param object The element to check for.
027         * @param targetPlayer The targeted player.
028         * @return Returns '{@code true}' if the element should be put in the view, '{@code false}' otherwise.
029         */
030        public boolean shouldBePlaced(InterfaceElements object, APlayer targetPlayer);
031        
032        /**
033         * Gets the position in the inventory view where the given {@linkplain InterfaceElements element} should be placed.
034         * 
035         * @param object The element to place.
036         * @param inventorySize The size of the inventory.
037         * @param targetPlayer The player being targeted.
038         * @return Returns the position of the given element or -1 if it should not be placed.
039         */
040        public int getPositionInInventory(InterfaceElements object, int inventorySize, APlayer targetPlayer);
041        
042        /**
043         * Gets the item stack to show to a player for a given {@linkplain InterfaceElements element}.
044         * 
045         * @param position The element being rendered.
046         * @param targetPlayer The targeted player.
047         * @param translator The I18n interface to handle translations.
048         * @return Returns the wrapped item stack to display.
049         */
050        public AnItem getItem(InterfaceElements position, APlayer targetPlayer, ITranslator translator);
051        
052        /**
053         * Draw the code (succession of digits) in the view.
054         * <p>
055         * The actual {@link Digits} are already converted to {@link InterfaceElements}.
056         * 
057         * @param inventory The inventory where the code should be displayed.
058         * @param values The elements composing the code.
059         * @param targetPlayer The targeted player.
060         * @param translator The I18n interface to handle translations.
061         */
062        public void drawCode(AnInventory inventory, InterfaceElements[] values, APlayer targetPlayer, ITranslator translator);
063        
064        /**
065         * This method is called when an error occur during the interface creation.
066         * 
067         * @param inventory The actual state of the inventory being created.
068         * @param targetPlayer The targeted player.
069         * @param translator The I18n interface to handle translations.
070         */
071        public void interfaceCreationFailure(AnInventory inventory, APlayer targetPlayer, ITranslator translator);
072}
