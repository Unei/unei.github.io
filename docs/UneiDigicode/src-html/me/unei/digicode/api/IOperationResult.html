005/**
006 * Represents a callback for a digicode operation.
007 * 
008 * @author JÃ«sFot
009 */
010@FunctionalInterface
011public interface IOperationResult
012{
013        /**
014         * Performs operations based on a digicode operation result.
015         * 
016         * @param codeIsCorrect '{@code true}' if the code given by the player matches, '{@code false}' otherwise.
017         * @param player The player who completed the operation.
018         * @param realCode The code requested to be typed by the player.
019         * @param typedCode The code the player has entered.
020         */
021        void run(boolean codeIsCorrect, APlayer player, String realCode, String typedCode);
022        
023        /**
024         * Called when a given player exits the view without completing the code.
025         * 
026         * @param player The player who cancelled the operation.
027         * @param realCode The correct code.
028         * @param currentCode The code the has typed right before canceling.
029         */
030        default void failed(APlayer player, String realCode, String currentCode) {};
031        
032        /**
033         * Called each times a given players enters or remove a digit in it's current view.
034         * 
035         * @param player The player in operation.
036         * @param realCode The code requested to be typed by the player.
037         * @param currentCode The actual code the player has given.
038         */
039        default void update(APlayer player, String realCode, String currentCode) {};
040}
