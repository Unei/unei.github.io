001package me.unei.digicode.api;
002
003/**
004 * Represent a code made of {@link me.unei.digicode.api.Digits}.
005 * 
006 * @author JÃ«sFot
007 */
008public interface ICode extends Iterable<Digits>, Cloneable
009{
010        /**
011         * Appends a digit to the end of this digit list (code).
012         * 
013         * @param digit The digit to add to this code.
014         * @return Returns `true` if this code has been modified, `false` otherwise.
015         */
016        boolean add(Digits digit);
017        
018        /**
019         * Sets a digit at a given location in the code.
020         * 
021         * @param pos The position where the digit should be set.
022         * @param digit The digit to set in this code.
023         * @return Returns `true` if this code has been modified, `false` otherwise.
024         */
025        boolean set(int pos, Digits digit);
026        
027        /**
028         * Reset this code to an empty one.
029         */
030        void reset();
031        
032        /**
033         * Gets the digit at the request position in the code.
034         * 
035         * @param pos The position where to get the digit from.
036         * @return Returns the digit at the given position or `null` if position is not in range.
037         */
038        Digits get(int pos);
039
040        /**
041         * Gets a full copy of the current code state.
042         * 
043         * @return Returns a copy of the code as a digit array.
044         */
045        Digits[] toRaw();
046        
047        /**
048         * Gets the current length of the code.
049         * 
050         * @return Returns the length of the code.
051         */
052        int getLength();
053        
054        /**
055         * Checks whenever the two codes represent the same digit list.
056         * 
057         * @param code The code to be compared with.
058         * @return Returns '{@code true}' if the 2 codes are same, '{@code false}' otherwise.
059         */
060        boolean isSame(ICode code);
061        
062        ICode clone();
063}
