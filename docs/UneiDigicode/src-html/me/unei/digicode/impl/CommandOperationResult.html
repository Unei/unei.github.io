001package me.unei.digicode.impl;
002
003import me.unei.digicode.api.IOperationResult;
004import me.unei.digicode.mp.APlayer;
005
006public class CommandOperationResult implements IOperationResult
007{
008        private final String successCommand;
009        private final String failureCommand;
010        
011        public CommandOperationResult(String success, String failure)
012        {
013                this.successCommand = success;
014                this.failureCommand = failure;
015        }
016        
017        @Override
018        public void run(boolean codeIsCorrect, APlayer player, String realCode, String typedCode)
019        {
020                if (codeIsCorrect)
021                {
022                        player.executeCommand(this.successCommand);
023                }
024                else
025                {
026                        player.executeCommand(this.failureCommand);
027                }
028        }
029}
