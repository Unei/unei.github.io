001package me.unei.digicode.impl;
002
003import java.io.Serializable;
004import java.util.Iterator;
005
006import me.unei.digicode.api.Digits;
007import me.unei.digicode.api.ICode;
008
009public class Code implements ICode, Serializable
010{
011        private static final long serialVersionUID = -3464770652476216220L;
012        
013        private Digits[] rawCode;
014        private int cursor = 0;
015        
016        public Code(int maxLength)
017        {
018                this.rawCode = new Digits[maxLength];
019        }
020        
021        Code(Digits[] array)
022        {
023                this.rawCode = array;
024                this.cursor = array.length;
025        }
026
027        @Override
028        public boolean add(Digits digit)
029        {
030                if (digit != null && cursor < rawCode.length)
031                {
032                        rawCode[cursor] = digit;
033                        cursor += 1;
034                        return true;
035                }
036                return false;
037        }
038
039        @Override
040        public boolean set(int pos, Digits digit)
041        {
042                if (pos >= 0 && pos < cursor)
043                {
044                        rawCode[pos] = digit;
045                        return true;
046                }
047                return false;
048        }
049
050        @Override
051        public void reset()
052        {
053                this.cursor = 0;
054        }
055        
056        public void decrement()
057        {
058                if (this.cursor > 0)
059                {
060                        this.cursor -= 1;
061                }
062        }
063
064        @Override
065        public Digits get(int pos)
066        {
067                if (pos >= 0 && pos < cursor)
068                {
069                        return rawCode[pos];
070                }
071                return null;
072        }
073        
074        @Override
075        public Digits[] toRaw()
076        {
077                Digits[] result = new Digits[cursor];
078                System.arraycopy(rawCode, 0, result, 0, cursor);
079                return result;
080        }
081        
082        @Override
083        public int getLength()
084        {
085                return cursor;
086        }
087        
088        @Override
089        public String toString()
090        {
091                char[] charCode = new char[cursor];
092                for (int i = 0; i < cursor; ++i)
093                {
094                        charCode[i] = rawCode[i].getCharCode();
095                }
096                return new String(charCode);
097        }
098        
099        @Override
100        public Code clone()
101        {
102                Code cp = new Code(rawCode.length);
103                System.arraycopy(rawCode, 0, cp.rawCode, 0, rawCode.length);
104                cp.cursor = this.cursor;
105                return cp;
106        }
107        
108        @Override
109        public boolean isSame(ICode other)
110        {
111                int i;
112                Digits copy[] = other.toRaw();
113                for (i = 0; i < Math.min(cursor, other.getLength()); ++i)
114                {
115                        if (rawCode[i] != copy[i])
116                        {
117                                return false;
118                        }
119                }
120                while (i < cursor)
121                {
122                        if (rawCode[i] != null)
123                        {
124                                return false;
125                        }
126                        i += 1;
127                }
128                while (i < other.getLength())
129                {
130                        if (copy[i] != null)
131                        {
132                                return false;
133                        }
134                        i += 1;
135                }
136                return true;
137        }
138
139        @Override
140        public Iterator<Digits> iterator()
141        {
142                return new DigitIterator(this);
143        }
144        
145        private static final class DigitIterator implements Iterator<Digits>
146        {
147                private int position;
148                private final Code parentInstance;
149                
150                public DigitIterator(Code parent)
151                {
152                        this.parentInstance = parent;
153                        this.position = 0;
154                }
155
156                @Override
157                public boolean hasNext()
158                {
159                        return this.position < this.parentInstance.getLength();
160                }
161
162                @Override
163                public Digits next()
164                {
165                        if (!hasNext())
166                        {
167                                return null;
168                        }
169                        Digits next = this.parentInstance.rawCode[this.position];
170                        this.position += 1;
171                        return next;
172                }
173        }
174
175        public static Code fromString(String string) {
176                Digits[] raw = new Digits[string.length()];
177                for (int i = 0; i < string.length(); ++i) {
178                        char c = string.charAt(i);
179                        Digits digit = Digits.getByChar(c);
180                        if (digit == null) {
181                                throw new NumberFormatException("Code should only contains numeric digits");
182                        }
183                        raw[i] = digit;
184                }
185                return new Code(raw);
186        }
187        
188        public static Code fromInt(int val) {
189                return fromString(Integer.toString(Math.abs(val)));
190        }
191}
