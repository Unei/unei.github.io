001package me.unei.digicode.impl;
002
003import me.unei.digicode.api.ITranslator;
004import me.unei.digicode.api.InterfaceElements;
005import me.unei.digicode.api.ThemeApplyer;
006import me.unei.digicode.config.MainConfigFile;
007import me.unei.digicode.mp.APlayer;
008import me.unei.digicode.mp.AnInventory;
009import me.unei.digicode.mp.AnItem;
010import me.unei.digicode.plugin.UneiDigit;
011
012public class DefaultThemeApplyer implements ThemeApplyer
013{
014        public static final int DIGICODE_INV_SIZE = 6 * 9;
015        
016        public static final int LOC_DIGIT_0 = 49;
017        public static final int LOC_DIGIT_1 = 21;
018        public static final int LOC_DIGIT_2 = 22;
019        public static final int LOC_DIGIT_3 = 23;
020        public static final int LOC_DIGIT_4 = 30;
021        public static final int LOC_DIGIT_5 = 31;
022        public static final int LOC_DIGIT_6 = 32;
023        public static final int LOC_DIGIT_7 = 39;
024        public static final int LOC_DIGIT_8 = 40;
025        public static final int LOC_DIGIT_9 = 41;
026        
027        public static final int LOC_EXIT = (DIGICODE_INV_SIZE - 1);
028        public static final int LOC_VALID = 36;
029        public static final int LOC_CANCEL = 18;
030        public static final int LOC_RETURN = 17;
031        
032        @Override
033        public int getInventorySize()
034        {
035                return DIGICODE_INV_SIZE;
036        }
037        
038        @Override
039        public boolean shouldBePlaced(InterfaceElements object, APlayer targetPlayer)
040        {
041                return object != null && object != InterfaceElements.Unknown;
042        }
043
044        @Override
045        public int getPositionInInventory(InterfaceElements object, int inventorySize, APlayer targetPlayer)
046        {
047                if (inventorySize < DIGICODE_INV_SIZE)
048                {
049                        return -1;
050                }
051                switch (object)
052                {
053                case Val_0: return LOC_DIGIT_0;
054                case Val_1: return LOC_DIGIT_1;
055                case Val_2: return LOC_DIGIT_2;
056                case Val_3: return LOC_DIGIT_3;
057                case Val_4: return LOC_DIGIT_4;
058                case Val_5: return LOC_DIGIT_5;
059                case Val_6: return LOC_DIGIT_6;
060                case Val_7: return LOC_DIGIT_7;
061                case Val_8: return LOC_DIGIT_8;
062                case Val_9: return LOC_DIGIT_9;
063
064                case Cancel: return LOC_CANCEL;
065                case Valid: return LOC_VALID;
066                case Return: return LOC_RETURN;
067                case Exit: return LOC_EXIT;
068                
069                default:
070                        return -1;
071                }
072        }
073
074        @Override
075        public AnItem getItem(InterfaceElements position, APlayer targetPlayer, ITranslator translator)
076        {
077                try
078                {
079                        switch (position)
080                        {
081                        case Exit: return UneiDigit.getInstance().createItem(MainConfigFile.exitItem);
082                        case Return: return UneiDigit.getInstance().createItem(MainConfigFile.returnItem);
083                        case Valid: return UneiDigit.getInstance().createItem(MainConfigFile.validateItem);
084                        case Cancel: return UneiDigit.getInstance().createItem(MainConfigFile.cancelItem);
085                        case Val_0: return UneiDigit.getInstance().createItem(MainConfigFile.val_0Item);
086                        case Val_1: return UneiDigit.getInstance().createItem(MainConfigFile.val_1Item);
087                        case Val_2: return UneiDigit.getInstance().createItem(MainConfigFile.val_2Item);
088                        case Val_3: return UneiDigit.getInstance().createItem(MainConfigFile.val_3Item);
089                        case Val_4: return UneiDigit.getInstance().createItem(MainConfigFile.val_4Item);
090                        case Val_5: return UneiDigit.getInstance().createItem(MainConfigFile.val_5Item);
091                        case Val_6: return UneiDigit.getInstance().createItem(MainConfigFile.val_6Item);
092                        case Val_7: return UneiDigit.getInstance().createItem(MainConfigFile.val_7Item);
093                        case Val_8: return UneiDigit.getInstance().createItem(MainConfigFile.val_8Item);
094                        case Val_9: return UneiDigit.getInstance().createItem(MainConfigFile.val_9Item);
095                        default: return null;
096                        }
097                }
098                catch (IllegalAccessException ise)
099                {
100                        return null;
101                }
102        }
103
104        @Override
105        public void drawCode(AnInventory inventory, InterfaceElements[] values, APlayer targetPlayer, ITranslator translator)
106        {
107                int length = Math.min(values.length, 9);
108                int startPoint = (4 - (int) (length / 2));
109                for (int i = 0; i < 9; ++i)
110                {
111                        if (i < startPoint)
112                        {
113                                inventory.removeAt(i);
114                                continue;
115                        }
116                        int pos = i - startPoint;
117                        if (pos >= values.length)
118                        {
119                                inventory.removeAt(i);
120                                continue;
121                        }
122                        InterfaceElements sel = values[pos];
123                        if (sel == null)
124                        {
125                                inventory.removeAt(i);
126                        }
127                        AnItem ai = getItem(sel, targetPlayer, translator);
128                        if (ai != null)
129                        {
130                                inventory.setItem(i, ai);
131                        }
132                }
133        }
134
135        @Override
136        public void interfaceCreationFailure(AnInventory inventory, APlayer targetPlayer, ITranslator translator)
137        {
138                targetPlayer.sendMessage("Could not open the digicode window");
139        }
140}
