001package me.unei.old_digicode.sample;
002
003import org.bukkit.command.Command;
004import org.bukkit.command.CommandSender;
005import org.bukkit.entity.Player;
006
007import me.unei.digicode.api.ICode;
008import me.unei.digicode.api.IDigicodeOperator;
009import me.unei.digicode.mp.APlayer;
010import me.unei.digicode.plugin.UneiDigit;
011
012public class SampleCode /* extends org.bukkit.plugin.java.JavaPlugin */
013{
014        public boolean onCommand(CommandSender sender, Command command, String label, String[] args)
015        {
016                // This is a sample code for the most basic usage of UneiDigit, there is no 'sample_command' in UneiDigit
017                if (command.getName().equalsIgnoreCase("sample_command"))
018                {
019                        try
020                        {
021                                UneiDigit inst = UneiDigit.getInstance(); // Get an instance of the plugin, will throw if not yet loaded.
022                                
023                                IDigicodeOperator operator = inst.getOperator(); // Get the digicode operations manager.
024                                
025                                APlayer target = inst.getPlayer(((Player) sender).getUniqueId()); // Get the player by it's UUID
026                                ICode code = UneiDigit.getCodeFromInt(1234); // Define the code the user must use.
027                                
028                                operator.startBasicOperation(target, code, null); // Start the operation and open the GUI to the player.
029                                
030                                // For personalized action upon operation completion (user validated the code) :
031                                operator.startBasicOperation(target, code, (codeIsCorrect, whoclicked, realCode, typedCode) -> {
032                                        // This method is called when the player is done with the GUI, whenever the code is right or not
033                                        // is indicated by parameter 'codeIsCorrect'.
034                                        if (codeIsCorrect) {
035                                                whoclicked.sendMessage("Code is good !");
036                                        }
037                                });
038                        }
039                        catch (IllegalAccessException e)
040                        {
041                                // UneiDigit is unexpectedly not yet loaded.
042                                // If you didn't, add the following line to your plugin.yml :
043                                // depend: [uneiDigiCode]
044                                e.printStackTrace();
045                        }
046                }
047                return true;
048        }
049}
