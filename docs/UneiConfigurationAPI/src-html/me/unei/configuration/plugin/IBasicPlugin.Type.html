001package me.unei.configuration.plugin;
002
003import java.io.File;
004import java.io.InputStream;
005import java.util.logging.Logger;
006
007public interface IBasicPlugin {
008
009    /**
010     * Returns the folder that the plugin data's files are located in. The folder may not yet exist.
011     * <p>
012     * See <tt>org.bukkit.plugin.Plugin#getDataFolder()</tt> within Bukkit<br>
013     * See <tt>net.md_5.bungee.api.plugin.Plugin#getDataFolder()</tt> within BungeeCord API
014     *
015     * @return the data folder of this plugin
016     */
017    public File getDataFolder();
018
019    /**
020     * Returns the plugin logger associated with this server's logger. The returned logger automatically tags all log messages with the plugin's name.
021     * <p>
022     * See <tt>org.bukkit.plugin.Plugin#getLogger()</tt> within Bukkit<br>
023     * See <tt>net.md_5.bungee.api.plugin.Plugin#getLogger()</tt> within BungeeCord API
024     *
025     * @return Logger associated with this plugin
026     */
027    public Logger getLogger();
028
029    /**
030     * Gets an embedded resource in this plugin, within the jar or container.
031     * Care must be taken to close the returned stream.
032     * <p>
033     * See <tt>org.bukkit.plugin.Plugin#getResource(String)</tt> within Bukkit<br>
034     * See <tt>net.md_5.bungee.api.plugin.Plugin#getResourceAsStream(String)</tt> within BungeeCord API
035     *
036     * @param path the full path name of this resource
037     *
038     * @return the stream for getting this resource, or null if it does not exist
039     */
040    public InputStream getResource(String path);
041
042    /**
043     * Returns the type of plugin this is.
044     *
045     * @return the type of plugin this is
046     */
047    public IBasicPlugin.Type getType();
048
049    /**
050     * Plugin types.
051     */
052    public static enum Type {
053
054        /**
055         * @see <a href="https://bukkit.org">https://bukkit.org</a>
056         */
057        BUKKIT,
058
059        /**
060         * @see <a href="https://www.spigotmc.org/wiki/bungeecord/">https://www.spigotmc.org/wiki/bungeecord/</a>
061         */
062        BUNGEECORD,
063
064        /**
065         * @see <a href="https://www.spongepowered.org">https://www.spongepowered.org</a>
066         */
067        SPONGE,
068
069        /**
070         * @see <a href="http://www.minecraftforge.net/forum/">http://www.minecraftforge.net/forum/</a>
071         */
072        FORGE,
073
074        /**
075         * @see <a href="https://en.wikipedia.org/wiki/Standalone_software">https://en.wikipedia.org/wiki/Standalone_software</a>
076         */
077        STANDALONE;
078
079    }
080}
