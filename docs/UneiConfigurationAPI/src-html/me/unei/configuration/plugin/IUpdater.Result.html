001package me.unei.configuration.plugin;
002
003import java.util.Date;
004
005/**
006 * Represent the updater behind UneiConfiguration.
007 */
008public interface IUpdater {
009
010        /**
011         * Synchronously check for a new version of UneiConfiguration.
012         * 
013         * @return {@link IUpdater.Result#UPDATE_AVAILABLE} if an update is available ;<br>
014         *  {@link IUpdater.Result#NO_UPDATE} if you the latest version ;<br>
015         *  {@link IUpdater.Result#FAILED} if the updater was unable to check for a newer version.
016         */
017        public IUpdater.Result checkVersion();
018
019        /**
020         * Asynchronously check for a new version of UneiConfiguration.
021         * 
022         * @see #checkVersion()
023         * 
024         * @param callback The callback witch will be called upon version check completion.
025         */
026        public void checkVersionAsync(final IUpdater.ICallback callback);
027        
028        /**
029         * Gets the current version string  of UneiConfiguration.
030         * 
031         * @return the current version.
032         */
033        public String getCurrentVersion();
034        
035        /**
036         * Gets the latest available version string of UneiConfiguration, if retrieved successfully.
037         * 
038         * @return The latest version string.
039         */
040        public String getLatestVersion();
041        
042        /**
043         * Gets the last time the online version of UneiConfiguration was updated.
044         * 
045         * @return the last update time.
046         */
047        public Date getLastUpdateTime();
048        
049        /**
050         * Gets the last time the online version of UneiConfiguration was updated.
051         * <p>
052         * This will return a pre-formatted string version of the date.
053         * 
054         * @see #getLastUpdateTime()
055         * 
056         * @return the last update time.
057         */
058        public String getReadableLastUpdateTime();
059        
060        /**
061         * Call back class used when calling the updater asynchronously.
062         */
063        public static interface ICallback {
064                
065                /**
066                 * Method called upon version checking completion.
067                 * 
068                 * @param updater The updater instance (if you want to re-run things).
069                 * @param result The update result.
070                 */
071                public void run(IUpdater updater, IUpdater.Result result);
072        }
073
074        /**
075         * Version checking result.
076         * <p>
077         * Could be :
078         * <ul>
079         * <li>{@linkplain Result#NO_UPDATE} - You are currently using the latest version available ;</li>
080         * <li>{@linkplain Result#UPDATE_AVAILABLE} - An update is available ;</li>
081         * <li>{@linkplain Result#FAILED} - The updater failed to retrieve online version</li>
082         * </ul>
083         */
084        public static enum Result {
085                
086                /**
087                 * The updater could get online version information,
088                 * but the currently running version is either same
089                 * or greater than the online one.
090                 * <p>
091                 * NO new version of UneiConfiguration is available.
092                 */
093                NO_UPDATE,
094                /**
095                 * The updater could get online version information,
096                 * and the currently running version is lesser than the online one.
097                 * <p>
098                 * An new version of UneiConfiguration is available. 
099                 */
100                UPDATE_AVAILABLE,
101                /**
102                 * The updater could not get online version information
103                 * and is not able to check for newer version.
104                 */
105                FAILED;
106        }
107}
