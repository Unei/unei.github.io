010        private static final byte[] EMPTY_BYTE_ARRAY = new byte[0];
011        private static final short[] EMPTY_SHORT_ARRAY = new short[0];
012        private static final int[] EMPTY_INT_ARRAY = new int[0];
013        private static final long[] EMPTY_LONG_ARRAY = new long[0];
014
015        public static Class<?> getIterableParam(Iterable<?> iterable) {
016                Class<?> result = null;
017                if (iterable == null) {
018                        return null;
019                }
020                Iterator<?> it = iterable.iterator();
021                if (it.hasNext()) {
022                        Object o = it.next();
023                        result = (o != null) ? o.getClass() : null;
024                }
025                return result;
026        }
027        
028        @SuppressWarnings("unchecked")
029        public static <T> T[] toArray(Iterable<T> value, Class<T> clz) {
030                if (value == null) {
031                        return null;
032                }
033                Collection<T> listed;
034                if (value instanceof Collection) {
035                        listed = (Collection<T>) value;
036                } else {
037                        listed = new ArrayList<>();
038                        Iterator<T> it = value.iterator();
039                        while (it.hasNext()) {
040                                listed.add(it.next());
041                        }
042                }
043                T[] result = (T[]) Array.newInstance(clz, listed.size());
044                return listed.toArray(result);
045        }
046
047        public static byte[] toPrimitive(Byte[] bytes) {
048                if (bytes == null || bytes.length <= 0) {
049                        return EMPTY_BYTE_ARRAY;
050                }
051                byte[] result = new byte[bytes.length];
052                for (int i = 0; i < bytes.length; ++i) {
053                        result[i] = bytes[i].byteValue();
054                }
055                return result;
056        }

091        public static byte[] toBytes(Object iterable)
092        {
093                if (iterable == null || !(iterable instanceof Iterable)) {
094                        return EMPTY_BYTE_ARRAY;
095                }
096                int length;
097                if (iterable instanceof Collection) {
098                        length = ((Collection<?>) iterable).size();
099                } else {
100                        int cnt = 0;
101                        Iterator<?> it = ((Iterable<?>) iterable).iterator();
102                        while (it.hasNext()) {
103                                cnt += 1;
104                        }
105                        length = cnt;
106                }
107                byte[] result = new byte[length];
108                int i = 0;
109                for (Iterator<?> it = ((Iterable<?>) iterable).iterator(); it.hasNext(); ++i) {
110                        result[i] = ((Number) it.next()).byteValue();
111                }
112                return result;
113        }
