001package me.unei.configuration.api;
002
003public interface IHRConfiguration extends IConfiguration {
004
005        /**
006         * Returns the representation as text of the data in this
007         * configuration, formatted as nicely as possible and in a
008         * way that is easy to read for a human.
009         *
010         * @return the formatted text representation of the data
011         */
012        public String toFormattedString();
013
014        /**
015         * Returns the representation as text of the data in this
016         * configuration, as short as possible and in a way that
017         * it is the smallest as possible, but still readable by
018         * a computer.
019         *
020         * @return the minimized text representation of the data
021         */
022        public String toMinimizedString();
023
024        /**
025         * Returns the representation as text of the data in this
026         * configuration, appointed for being saved inside a file
027         * and without indication on whether it should be formatted
028         * or minimized.
029         * <p>
030         * Usually, will be the formatted version to allow humans
031         * to read the contents of the file easily, but might be
032         * a minimized version or even one between both.
033         *
034         * @return the text representation of the data
035         */
036        public String saveToString();
037
038        /**
039         * Reads the required data from the specified string.
040         *
041         * The string can be the result of <tt>saveToString</tt>,
042         * <tt>toFormattedString</tt> or <tt>toMinimizedString</tt>,
043         * or can come from any other source provided that its format
044         * is correct and readable by a computer.
045         *
046         * @param data the text representation of the data
047         */
048        public void loadFromString(String data);
049}
