001package me.unei.configuration.api;
002
003import java.util.List;
004
005public interface IFlatCSVConfiguration extends IFlatConfiguration {
006
007        /**
008         * Returns a {@link List List} of lines located at the top of the file,
009         * which are not part of the data, and that contains the different keys.
010         *
011         * @return the {@link List List} of lines at the top of the file
012         */
013        public List<String> getHeaderLine();
014
015        /**
016         * Resets the {@link List List} of lines located at the top of the file,
017         * which are not part of the data, and that contains the different keys,
018         * to the default header.
019         *
020         * Usually, the default header will contain a first value <tt>Key</tt> and
021         * a second one <tt>Value</tt> separated by a comma, but this may change
022         * over time and/or in different implementations.
023         */
024        public void resetHeaderLine();
025
026        /**
027         * Sets the value relative to the specified key to the specified
028         * {@link String String} {@link List List}.
029         *
030         * @param key the key to give to the specified value
031         * @param value the {@link String String} {@link List List} to set relative to the specified key
032         */
033        public void setList(String key, List<String> value);
034
035        /**
036         * Returns the value relative to the specified key as {@link String String} {@link List List},
037         * if possible.
038         *
039         * @param key the key of the value to retrieve
040         * @return the {@link String String} {@link List List} relative to the specified key
041         */
042        public List<String> getList(String key);
043}
