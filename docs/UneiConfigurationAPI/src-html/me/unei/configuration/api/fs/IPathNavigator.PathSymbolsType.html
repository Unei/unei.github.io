001package me.unei.configuration.api.fs;
002
003import java.util.function.Consumer;
004
005import me.unei.configuration.api.fs.IPathComponent.IPathComponentsList;
006
007/**
008 * Section navigator.
009 *
010 * @param <T> The section type.
011 */
012public interface IPathNavigator<T extends NavigableFile> {
013
014        /**
015         * Navigate to the root section.
016         */
017        public void goToRoot();
018
019        /**
020         * Navigate to the parent section.
021         */
022        public void goToParent();
023
024        /**
025         * Navigate to the `name` child section.
026         * 
027         * @param name The name of the child section to go to.
028         */
029        public void goToChild(String name);
030
031        /**
032         * Navigate to the `index` element section.
033         * 
034         * @param index The index of the child section to go to.
035         */
036        public void goToIndex(int index);
037
038        /**
039         * Gets the current path.
040         * 
041         * @return Returns the path as a string.
042         */
043        public String getCurrentPath();
044
045        /**
046         * Gets the current section.
047         * 
048         * @return Returns the actual node.
049         */
050        public T getCurrentNode();
051
052        /**
053         * Navigate following the given path and execute an action on each nodes.
054         * 
055         * @param path The path of the destination.
056         * @param action An action to execute on each node of the path.
057         * 
058         * @throws NullPointerException If the `action` parameter is null.
059         */
060        public void followAndApply(IPathComponentsList path, Consumer<T> action);
061
062        /**
063         * Navigate following the given `path`.
064         * 
065         * @param path The path of the destination.
066         * @return Returns `true` if navigation was successful, `false` otherwise.
067         * 
068         * @see #navigate(String, PathSymbolsType)
069         */
070        public boolean followPath(IPathComponentsList path);
071
072        /**
073         * Navigate following the given `path`.
074         * 
075         * @param path The path of the destination.
076         * @param type The types of the symbols used in the path.
077         * @return Returns `true` if navigation was successful, `false` otherwise.
078         * 
079         * @see #followPath(IPathComponentsList)
080         */
081        public boolean navigate(String path, PathSymbolsType type);
082
083        /**
084         * Represents the different types of symbols used in string paths.
085         */
086        public static enum PathSymbolsType {
087                /**
088                 * The default Bukkit path symbols type.
089                 * 
090                 * <ul>
091                 * <li>Escape char: &nbsp; &nbsp; &nbsp;                        <tt>'\'</tt></li>
092                 * <li>Separator char:&nbsp;                                            <tt>'.'</tt></li>
093                 * <li>Root char: &nbsp; &nbsp; &nbsp; &nbsp; &nbsp;<tt>'.'</tt></li>
094                 * <li>Parent string: &nbsp; &nbsp;                             <tt>'..'</tt></li>
095                 * <li>Index prefix: &nbsp; &nbsp; &nbsp;                       <tt>'['</tt></li>
096                 * <li>Index suffix: &nbsp; &nbsp; &nbsp;&nbsp;         <tt>']'</tt></li>
097                 * </ul>
098                 */
099                BUKKIT('\\', '.', '.', "..", '[', ']', false),
100                /**
101                 * The default Unix path symbols type.
102                 * 
103                 * <ul>
104                 * <li>Escape char: &nbsp; &nbsp; &nbsp;                        <tt>'\'</tt></li>
105                 * <li>Separator char:&nbsp;                                            <tt>'/'</tt></li>
106                 * <li>Root char: &nbsp; &nbsp; &nbsp; &nbsp; &nbsp;<tt>'/'</tt></li>
107                 * <li>Parent string: &nbsp; &nbsp;                             <tt>'..'</tt></li>
108                 * <li>Index prefix: &nbsp; &nbsp; &nbsp;                       <tt>'['</tt></li>
109                 * <li>Index suffix: &nbsp; &nbsp; &nbsp;&nbsp;         <tt>']'</tt></li>
110                 * </ul>
111                 */
112                UNIX('\\', '/', '/', "..", '[', ']', true);
113
114                public final char escape;
115                public final char separator;
116                public final char root;
117                public final String parent;
118                public final char indexerPrefix;
119                public final char indexerSuffix;
120
121                public final boolean wrapParent;
122
123                private PathSymbolsType(char p_escape, char p_separator, char p_root, String p_parent, char idxPre, char idxSuf, boolean wParent) {
124                        this.escape = p_escape;
125                        this.separator = p_separator;
126                        this.root = p_root;
127                        this.parent = p_parent;
128                        this.indexerPrefix = idxPre;
129                        this.indexerSuffix = idxSuf;
130                        this.wrapParent = wParent;
131                }
132
133                /**
134                 * Try to detect the type of symbols used in the given 'apath'.
135                 * 
136                 * @param apath A path to test.
137                 * @return Returns a PathSymbolsType detected in the path.
138                 * 
139                 * @deprecated This won't detect the type for sure and can lead to data-reading errors.
140                 */
141                @Deprecated
142                public static PathSymbolsType tryDetectType(String apath) {
143                        int li = apath.lastIndexOf(UNIX.parent);
144                        if (li >= 0) {
145                                if (li > 0 && apath.charAt(li - 1) == UNIX.separator) {
146                                        return PathSymbolsType.UNIX;
147                                } else if ((li + 2) < apath.length() && apath.charAt(li + 2) == UNIX.separator) {
148                                        return PathSymbolsType.UNIX;
149                                }
150                        }
151                        return PathSymbolsType.BUKKIT;
152                }
153        }
154}
