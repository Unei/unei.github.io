001package me.unei.configuration.api.fs;
002
003/**
004 * Represent a section with a parent section and a child section.
005 */
006public interface NavigableFile {
007
008    /**
009     * Returns the path to get from this navigable file's
010     * root to the current node.
011     *
012     * @return the current path of this navigable file
013     */
014    public IPathComponent.IPathComponentsList getRealListPath();
015
016    /**
017     * Returns the path to get from this navigable file's
018     * root to the current node.
019     *
020     * @return the current path of this navigable file
021     */
022    public String getCurrentPath();
023
024    /**
025     * Returns this navigable file's direct (first) parent,
026     * if it has one, or itself if it doesn't have any parent.
027     *
028     * @return this navigable file's direct parent
029     */
030    public NavigableFile getParent();
031
032    /**
033     * Gets this navigable file's root (last parent).
034     * It returns itself if it doesn't have any parent.
035     *
036     * @return this navigable file's root
037     */
038    public NavigableFile getRoot();
039
040    /**
041     * Returns the child of this navigable file with the
042     * provided name, or <tt>null</tt> if it doesn't exist
043     * and if it can't be created.
044     *
045     * @param name The name of the child to return
046     * @return this navigable file's child with the provided name
047     */
048    public NavigableFile getChild(String name);
049
050    /**
051     * Returns the child of this navigable file at the
052     * provided index, or <tt>null</tt> if it doesn't exist
053     * and if it can't be created.
054     *
055     * @param index The index of the child to return
056     * @return this navigable file's child at the provided index
057     */
058    public NavigableFile getAt(int index);
059}
