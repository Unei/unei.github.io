001package me.unei.configuration.api;
002
003import me.unei.configuration.api.format.INBTTag;
004
005public interface INBTConfiguration extends IConfiguration {
006
007    /**
008     * Returns a copy of the tag that represents this configuration
009     * and therefore containing all the data in this configuration.
010     *
011     * @return a copy of the tag that represents this configuration
012     */
013    public INBTTag getTagCopy();
014
015    /**
016     * Sets the current configuration representation to a
017     * copy of the specified {@link INBTTag tag}.
018     *
019     * @param tag the tag from which a copy will be made
020     */
021    public void setTagCopy(INBTTag tag);
022}
