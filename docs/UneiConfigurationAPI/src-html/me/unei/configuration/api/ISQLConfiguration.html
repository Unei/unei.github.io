001package me.unei.configuration.api;
002
003import java.io.Closeable;
004import java.sql.ResultSet;
005import java.sql.SQLException;
006import java.util.Map;
007
008public interface ISQLConfiguration extends Closeable {
009
010        /**
011         * Gets the name of the table this section is in.
012         * 
013         * @return the name of the SQL table.
014         */
015        public String getTableName();
016        
017    /**
018     * Attempts to reconnect to the remote database, or to reopen the file.
019     *
020     * @throws SQLException if a database access error occurs
021     */
022    public void reconnect() throws SQLException;
023
024    /**
025     * Executes the specified SQL query after applying the specified parameters.
026     *
027     * @see java.sql.PreparedStatement#execute()
028     *
029     * @param query the SQL query to prepare and then execute
030     * @param parameters a map of the parameters to apply to the query
031     *
032     * @return <code>true</code> if the first result is a <code>ResultSet</code>
033     *         object; <code>false</code> if the first result is an update
034     *         count or there is no result
035     *
036     * @throws SQLException if a database access error occurs
037     */
038    public boolean execute(String query, Map<Integer, Object> parameters) throws SQLException;
039
040    /**
041     * Executes the specified SQL query after applying the specified parameters,
042     * and returns the {@link ResultSet result}.
043     *
044     * @see java.sql.PreparedStatement#executeQuery()
045     *
046     * @param query the SQL query to prepare and then execute
047     * @param parameters a map of the parameters to apply to the query
048     *
049     * @return a <code>ResultSet</code> object that contains the data produced by the
050     *         query; never <code>null</code>
051     *
052     * @throws SQLException if a database access error occurs
053     */
054    public ResultSet query(String query, Map<Integer, Object> parameters) throws SQLException;
055
056    /**
057     * Executes the specified SQL query after applying the specified parameters,
058     * and returns the integer returned in case of an update.
059     *
060     * @see java.sql.PreparedStatement#executeUpdate()
061     *
062     * @param query the SQL query to prepare and then execute
063     * @param parameters a map of the parameters to apply to the query
064     *
065     * @return either (1) the row count for SQL Data Manipulation Language (DML) statements
066     *         or (2) 0 for SQL statements that return nothing
067     *
068     * @throws SQLException if a database access error occurs
069     */
070    public int update(String query, Map<Integer, Object> parameters) throws SQLException;
071
072    /**
073     * Executes the specified SQL query after applying the specified parameters,
074     * and returns the long returned in case of an update.
075     *
076     * @see java.sql.PreparedStatement#executeLargeUpdate()
077     *
078     * @param query the SQL query to prepare and then execute
079     * @param parameters a map of the parameters to apply to the query
080     *
081     * @return either (1) the row count for SQL Data Manipulation Language
082     *         (DML) statements or (2) 0 for SQL statements that return nothing
083     *
084     * @throws SQLException if a database access error occurs
085     */
086    public long largeUpdate(String query, Map<Integer, Object> parameters) throws SQLException;
087}
