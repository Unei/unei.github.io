001package me.unei.configuration.api.exceptions;
002
003import java.io.File;
004
005/**
006 * An exception thrown when a file format is not the one expected.
007 */
008public class FileFormatException extends Exception
009{
010        private static final long serialVersionUID = 3594972910118569236L;
011
012    /** Constructs a new runtime exception with the specified detail message.
013     * The cause is not initialized, and may subsequently be initialized by a
014     * call to {@link #initCause}.
015     *
016     * @param   message   the detail message. The detail message is saved for
017     *          later retrieval by the {@link #getMessage()} method.
018     */
019        FileFormatException(String message)
020        {
021                super(message);
022        }
023
024    /**
025     * Constructs a new runtime exception with the specified detail message and
026     * cause.  <p>Note that the detail message associated with
027     * {@code cause} is <i>not</i> automatically incorporated in
028     * this runtime exception's detail message.
029     *
030     * @param  message the detail message (which is saved for later retrieval
031     *         by the {@link #getMessage()} method).
032     * @param  t the cause (which is saved for later retrieval by the
033     *         {@link #getCause()} method).  (A <tt>null</tt> value is
034     *         permitted, and indicates that the cause is nonexistent or
035     *         unknown.)
036     */
037        FileFormatException(String message, Throwable t)
038        {
039                super(message, t);
040        }
041
042    /** Constructs a new runtime exception with the specified detail message.
043     * The cause is not initialized, and may subsequently be initialized by a
044     * call to {@link #initCause}.
045         * 
046         * @param requestedFormat The format expected from the file.
047         * @param file The file in the wrong format (cause of this error).
048     * @param   message   the detail message. The detail message is saved for
049     *          later retrieval by the {@link #getMessage()} method.
050         */
051        public FileFormatException(String requestedFormat, File file, String message)
052        {
053                super("The file " + file.getAbsolutePath() + " is not in format " + requestedFormat + " : " + message);
054        }
055
056    /**
057     * Constructs a new runtime exception with the specified detail message and
058     * cause.  <p>Note that the detail message associated with
059     * {@code cause} is <i>not</i> automatically incorporated in
060     * this runtime exception's detail message.
061         * 
062         * @param requestedFormat The format expected from the file.
063         * @param file The file in the wrong format (cause of this error).
064     * @param  message the detail message (which is saved for later retrieval
065     *         by the {@link #getMessage()} method).
066     * @param  t the cause (which is saved for later retrieval by the
067     *         {@link #getCause()} method).  (A <tt>null</tt> value is
068     *         permitted, and indicates that the cause is nonexistent or
069     *         unknown.)
070         */
071        public FileFormatException(String requestedFormat, File file, String message, Throwable t)
072        {
073                super("The file " + file.getAbsolutePath() + " is not in format " + requestedFormat + " : " + message, t);
074        }
075}
