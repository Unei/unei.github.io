001package me.unei.configuration.api.exceptions;
002
003/**
004 * An exception thrown when an instance is not of an expected class type.
005 */
006public class UnexpectedClassException extends RuntimeException
007{
008        private static final long serialVersionUID = -8302500070768143554L;
009
010    /** Constructs a new runtime exception with {@code null} as its
011     * detail message.  The cause is not initialized, and may subsequently be
012     * initialized by a call to {@link #initCause}.
013     */
014        public UnexpectedClassException()
015        {
016                super();
017        }
018
019    /** Constructs a new runtime exception with the specified detail message.
020     * The cause is not initialized, and may subsequently be initialized by a
021     * call to {@link #initCause}.
022     *
023     * @param   message   the detail message. The detail message is saved for
024     *          later retrieval by the {@link #getMessage()} method.
025     */
026        public UnexpectedClassException(String message)
027        {
028                super(message);
029        }
030
031    /**
032     * Constructs a new runtime exception with the specified detail message and
033     * cause.  <p>Note that the detail message associated with
034     * {@code cause} is <i>not</i> automatically incorporated in
035     * this runtime exception's detail message.
036     *
037     * @param  message the detail message (which is saved for later retrieval
038     *         by the {@link #getMessage()} method).
039     * @param  cause the cause (which is saved for later retrieval by the
040     *         {@link #getCause()} method).  (A <tt>null</tt> value is
041     *         permitted, and indicates that the cause is nonexistent or
042     *         unknown.)
043     */
044        public UnexpectedClassException(String message, Throwable cause)
045        {
046                super(message, cause);
047        }
048        
049        public UnexpectedClassException(Class<?> obtained)
050        {
051                super("Class of type " + obtained.getTypeName() + " was not expected here !");
052        }
053        
054        public UnexpectedClassException(Class<?> obtained, Throwable cause)
055        {
056                super("Class of type " + obtained.getTypeName() + " was not expected here !", cause);
057        }
058        
059        public UnexpectedClassException(Class<?> expected, Class<?> obtained)
060        {
061                super("Expected class of type " + expected.getTypeName() + " but got class of type " + obtained.getTypeName() + " !");
062        }
063        
064        public UnexpectedClassException(Class<?> expected, Class<?> obtained, Throwable cause)
065        {
066                super("Expected class of type " + expected.getTypeName() + " but got class of type " + obtained.getTypeName() + " !", cause);
067        }
068}
