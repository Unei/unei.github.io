001package me.unei.configuration.api.exceptions;
002
003import me.unei.configuration.SavedFile;
004
005/**
006 * An exception thrown when an element of a configuration instance could not be found.
007 * 
008 * @see me.unei.configuration.api.IFlatConfiguration#contains(String)
009 * 
010 * @author JÃ«sFot
011 * @since 1.5
012 * @version 1.0
013 */
014public class NoFieldException extends Exception {
015        private static final long serialVersionUID = 5787167801078410025L;
016
017        /** Constructs a new exception with the specified detail message.
018         * The cause is not initialized, and may subsequently be initialized by a
019         * call to {@link #initCause}.
020         *
021         * @param   message   the detail message. The detail message is saved for
022         *          later retrieval by the {@link #getMessage()} method.
023         */
024        NoFieldException(String message) {
025                super(message);
026        }
027
028        /**
029         * Constructs a new exception with the specified detail message and
030         * cause.  <p>Note that the detail message associated with
031         * {@code cause} is <i>not</i> automatically incorporated in
032         * this exception's detail message.
033         *
034         * @param  message the detail message (which is saved for later retrieval
035         *         by the {@link #getMessage()} method).
036         * @param  t the cause (which is saved for later retrieval by the
037         *         {@link #getCause()} method).  (A <tt>null</tt> value is
038         *         permitted, and indicates that the cause is nonexistent or
039         *         unknown.)
040         */
041        NoFieldException(String message, Throwable t) {
042                super(message, t);
043        }
044
045        /** Constructs a new no field exception with the specified detail message.
046         * The cause is not initialized, and may subsequently be initialized by a
047         * call to {@link #initCause}.
048         * 
049         * @param requestedKey The key of the field that is not available.
050         * @param file The file in i witch the key does not exists.
051         * @param message The detail message. It is added to the end of the generated message.
052         */
053        public NoFieldException(String requestedKey, SavedFile file, String message) {
054                super("The key " + requestedKey + " does not exists in file " + file.getFullName() + " : " + message);
055        }
056}
