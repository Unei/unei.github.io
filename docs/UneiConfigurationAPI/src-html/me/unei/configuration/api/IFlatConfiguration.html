001package me.unei.configuration.api;
002
003import java.util.Set;
004
005import me.unei.configuration.SavedFile;
006import me.unei.configuration.api.Configurations.ConfigurationType;
007import me.unei.configuration.api.exceptions.FileFormatException;
008
009public interface IFlatConfiguration {
010
011        /**
012         * Get the type of this configuration node instance.
013         * 
014         * @return this configuration's type.
015         */
016        public ConfigurationType getConfigurationType();
017        
018    /**
019     * Returns the {@link SavedFile SavedFile} that represents this
020     * configuration's file location.
021     *
022     * @return this configuration's {@link SavedFile SavedFile}
023     */
024    public SavedFile getFile();
025
026    /**
027     * Returns the name of the {@link SavedFile SavedFile} that
028     * represents this configuration's file location.
029     *
030     * @return the name of this configuration's {@link SavedFile SavedFile}
031     */
032    public String getFileName();
033
034    /**
035     * Returns the name of this configuration.
036     *
037     * Usually equivalent to {@link IFlatConfiguration#getFileName()}.
038     *
039     * @return the name of this configuration's {@link SavedFile SavedFile}
040     */
041    public String getName();
042
043    /**
044     * Returns <tt>true</tt> if it is currently possible to access and
045     * modify this configuration.
046     *
047     * @return <tt>true</tt> if this configuration is accessible
048     */
049    public boolean canAccess();
050
051    /**
052     * Locks this configuration and the file that represents it if applicable.
053     */
054    public void lock();
055
056    /**
057     * Saves all the modifications made to this configuration to the
058     * appropriate location.
059     *
060     * Usually saves this configuration's data to its file location.
061     */
062    public void save();
063
064    /**
065     * Reloads this configuration contents with the one stored, deleting
066     * any unsaved modification.
067     *
068     * @throws FileFormatException if the format of the read file is invalid
069     */
070    public void reload() throws FileFormatException;
071
072    /**
073     * Returns a set of the keys of all the direct children of this configuration.
074     *
075     * @return a set of the keys of all the direct children of this configuration
076     */
077    public Set<String> getKeys();
078
079    /**
080     * Returns <tt>true</tt> if this configuration contains a direct child with
081     * the specified name.
082     *
083     * @param key the name of the child to check
084     * @return <tt>true</tt> if this configuration contains a direct child with the specified name
085     */
086    public boolean contains(String key);
087
088    /**
089     * Returns the direct child element of this configuration which has
090     * the specified key as name as {@link String String}, if possible.
091     *
092     * @param key the name of the child to retrieve
093     * @return the direct child element with the specified name as {@link String String}
094     */
095    public String getString(String key);
096
097    /**
098     * Returns the direct child element of this configuration which has
099     * the specified key as name as double, if possible.
100     *
101     * @param key the name of the child to retrieve
102     * @return the direct child element with the specified name as double
103     */
104    public double getDouble(String key);
105
106    /**
107     * Returns the direct child element of this configuration which has
108     * the specified key as name as boolean, if possible.
109     *
110     * @param key the name of the child to retrieve
111     * @return the direct child element with the specified name as boolean
112     */
113    public boolean getBoolean(String key);
114
115    /**
116     * Returns the direct child element of this configuration which has
117     * the specified key as name as byte, if possible.
118     *
119     * @param key the name of the child to retrieve
120     * @return the direct child element with the specified name as byte
121     */
122    public byte getByte(String key);
123
124    /**
125     * Returns the direct child element of this configuration which has
126     * the specified key as name as short, if possible.
127     *
128     * @param key the name of the child to retrieve
129     * @return the direct child element with the specified name as short
130     */
131    public short getShort(String key);
132
133    /**
134     * Returns the direct child element of this configuration which has
135     * the specified key as name as float, if possible.
136     *
137     * @param key the name of the child to retrieve
138     * @return the direct child element with the specified name as float
139     */
140    public float getFloat(String key);
141
142    /**
143     * Returns the direct child element of this configuration which has
144     * the specified key as name as int, if possible.
145     *
146     * @param key the name of the child to retrieve
147     * @return the direct child element with the specified name as int
148     */
149    public int getInteger(String key);
150
151    /**
152     * Returns the direct child element of this configuration which has
153     * the specified key as name as long, if possible.
154     *
155     * @param key the name of the child to retrieve
156     * @return the direct child element with the specified name as long
157     */
158    public long getLong(String key);
159
160    /**
161     * Sets the specified {@link String String} as a child of this configuration with
162     * the specified key as name.
163     *
164     * @param key the name to give to the child
165     * @param value the {@link String String} to set as child
166     */
167    public void setString(String key, String value);
168
169    /**
170     * Sets the specified double as a child of this configuration with
171     * the specified key as name.
172     *
173     * @param key the name to give to the child
174     * @param value the double to set as child
175     */
176    public void setDouble(String key, double value);
177
178    /**
179     * Sets the specified boolean as a child of this configuration with
180     * the specified key as name.
181     *
182     * @param key the name to give to the child
183     * @param value the boolean to set as child
184     */
185    public void setBoolean(String key, boolean value);
186
187    /**
188     * Sets the specified byte as a child of this configuration with
189     * the specified key as name.
190     *
191     * @param key the name to give to the child
192     * @param value the byte to set as child
193     */
194    public void setByte(String key, byte value);
195
196    /**
197     * Sets the specified short as a child of this configuration with
198     * the specified key as name.
199     *
200     * @param key the name to give to the child
201     * @param value the short to set as child
202     */
203    public void setShort(String key, short value);
204
205    /**
206     * Sets the specified float as a child of this configuration with
207     * the specified key as name.
208     *
209     * @param key the name to give to the child
210     * @param value the float to set as child
211     */
212    public void setFloat(String key, float value);
213
214    /**
215     * Sets the specified int as a child of this configuration with
216     * the specified key as name.
217     *
218     * @param key the name to give to the child
219     * @param value the int to set as child
220     */
221    public void setInteger(String key, int value);
222
223    /**
224     * Sets the specified long as a child of this configuration with
225     * the specified key as name.
226     *
227     * @param key the name to give to the child
228     * @param value the long to set as child
229     */
230    public void setLong(String key, long value);
231
232    /**
233     * Removes any direct child element of this configuration which has
234     * the specified key as name.
235     *
236     * @param key the name of the child to remove
237     */
238    public void remove(String key);
239}
