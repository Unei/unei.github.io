001package me.unei.configuration.api.format;
002
003import java.util.Set;
004
005/**
006 * Represents a NBT Compound Tag.
007 */
008public interface INBTCompound extends INBTTag {
009
010    /**
011     * Returns a set of the keys of all the direct children of this tag.
012     *
013     * @return a set of the keys of all the direct children of this tag
014     */
015    public Set<String> keySet();
016
017    /**
018     * Returns the number of direct children of this tag.
019     *
020     * @return the number of direct children of this tag
021     */
022    public int size();
023
024    /**
025     * Sets the specified element as a child of this tag with
026     * the specified key as name.
027     *
028     * @param key the name to give to the child
029     * @param elem the element to set as child
030     */
031    public void set(String key, INBTTag elem);
032
033    /**
034     * Sets the specified byte as a child of this tag with
035     * the specified key as name.
036     *
037     * @param key the name to give to the child
038     * @param value the byte to set as child
039     */
040    public void setByte(String key, byte value);
041
042    /**
043     * Sets the specified short as a child of this tag with
044     * the specified key as name.
045     *
046     * @param key the name to give to the child
047     * @param value the short to set as child
048     */
049    public void setShort(String key, short value);
050
051    /**
052     * Sets the specified int as a child of this tag with
053     * the specified key as name.
054     *
055     * @param key the name to give to the child
056     * @param value the int to set as child
057     */
058    public void setInt(String key, int value);
059
060    /**
061     * Sets the specified long as a child of this tag with
062     * the specified key as name.
063     *
064     * @param key the name to give to the child
065     * @param value the long to set as child
066     */
067    public void setLong(String key, long value);
068
069    /**
070     * Sets the specified float as a child of this tag with
071     * the specified key as name.
072     *
073     * @param key the name to give to the child
074     * @param value the float to set as child
075     */
076    public void setFloat(String key, float value);
077
078    /**
079     * Sets the specified double as a child of this tag with
080     * the specified key as name.
081     *
082     * @param key the name to give to the child
083     * @param value the double to set as child
084     */
085    public void setDouble(String key, double value);
086
087    /**
088     * Sets the specified {@link String String} as a child of this tag with
089     * the specified key as name.
090     *
091     * @param key the name to give to the child
092     * @param value the {@link String String} to set as child
093     */
094    public void setString(String key, String value);
095
096    /**
097     * Sets the specified byte array as a child of this tag with
098     * the specified key as name.
099     *
100     * @param key the name to give to the child
101     * @param value the byte array to set as child
102     */
103    public void setByteArray(String key, byte[] value);
104
105    /**
106     * Sets the specified int array as a child of this tag with
107     * the specified key as name.
108     *
109     * @param key the name to give to the child
110     * @param value the int array to set as child
111     */
112    public void setIntArray(String key, int[] value);
113
114    /**
115     * Sets the specified long array as a child of this tag with
116     * the specified key as name.
117     *
118     * @param key the name to give to the child
119     * @param value the long array to set as child
120     */
121    public void setLongArray(String key, long[] value);
122
123    /**
124     * Sets the specified boolean as a child of this tag with
125     * the specified key as name.
126     *
127     * @param key the name to give to the child
128     * @param value the boolean to set as child
129     */
130    public void setBoolean(String key, boolean value);
131
132    /**
133     * Returns the direct child element of this tag which has
134     * the specified key as name.
135     *
136     * @param key the name of the child to retrieve
137     * @return the direct child element with the specified name
138     */
139    public INBTTag get(String key);
140
141    /**
142     * Returns the type as byte of the child element of this tag
143     * which has the specified key as name.
144     *
145     * @deprecated Magic value. prefer using {@link #getTypeOfTag(String)}.
146     * @param key the name of the child from which to retrieve the type
147     * @return the type as byte of the direct child element with the specified name
148     */
149    @Deprecated
150    public byte getTypeOf(String key);
151    
152    /**
153     * Returns the tag type of the child element of this tag
154     * which has the specified key as name.
155     *
156     * @param key the name of the child from which to retrieve the type
157     * @return the tag type of the direct child element with the specified name
158     */
159    public default TagType getTypeOfTag(String key) {
160        return TagType.getByTypeId(getTypeOf(key));
161    }
162
163    /**
164     * Returns <tt>true</tt> if this tag contains a direct child with
165     * the specified name.
166     *
167     * @param key the name of the child to check
168     * @return <tt>true</tt> if this tag contains a direct child with the specified name
169     */
170    public boolean hasKey(String key);
171
172    /**
173     * Returns the direct child element of this tag which has
174     * the specified key as name as byte, if possible.
175     *
176     * @param key the name of the child to retrieve
177     * @return the direct child element with the specified name as byte
178     */
179    public byte getByte(String key);
180
181    /**
182     * Returns the direct child element of this tag which has
183     * the specified key as name as short, if possible.
184     *
185     * @param key the name of the child to retrieve
186     * @return the direct child element with the specified name as short
187     */
188    public short getShort(String key);
189
190    /**
191     * Returns the direct child element of this tag which has
192     * the specified key as name as int, if possible.
193     *
194     * @param key the name of the child to retrieve
195     * @return the direct child element with the specified name as int
196     */
197    public int getInt(String key);
198
199    /**
200     * Returns the direct child element of this tag which has
201     * the specified key as name as long, if possible.
202     *
203     * @param key the name of the child to retrieve
204     * @return the direct child element with the specified name as long
205     */
206    public long getLong(String key);
207
208    /**
209     * Returns the direct child element of this tag which has
210     * the specified key as name as float, if possible.
211     *
212     * @param key the name of the child to retrieve
213     * @return the direct child element with the specified name as float
214     */
215    public float getFloat(String key);
216
217    /**
218     * Returns the direct child element of this tag which has
219     * the specified key as name as double, if possible.
220     *
221     * @param key the name of the child to retrieve
222     * @return the direct child element with the specified name as double
223     */
224    public double getDouble(String key);
225
226    /**
227     * Returns the direct child element of this tag which has
228     * the specified key as name as {@link String String}, if possible.
229     *
230     * @param key the name of the child to retrieve
231     * @return the direct child element with the specified name as {@link String String}
232     */
233    public String getString(String key);
234
235    /**
236     * Returns the direct child element of this tag which has
237     * the specified key as byte array as byte, if possible.
238     *
239     * @param key the name of the child to retrieve
240     * @return the direct child element with the specified name as byte array
241     */
242    public byte[] getByteArray(String key);
243
244    /**
245     * Returns the direct child element of this tag which has
246     * the specified key as name as int array, if possible.
247     *
248     * @param key the name of the child to retrieve
249     * @return the direct child element with the specified name as int array
250     */
251    public int[] getIntArray(String key);
252
253    /**
254     * Returns the direct child element of this tag which has
255     * the specified key as name as long array, if possible.
256     *
257     * @param key the name of the child to retrieve
258     * @return the direct child element with the specified name as long array
259     */
260    public long[] getLongArray(String key);
261
262    /**
263     * Returns the direct child element of this tag which has
264     * the specified key as name as an {@link INBTCompound INBTCompound}, if possible.
265     *
266     * @param key the name of the child to retrieve
267     * @return the direct child element with the specified name as an {@link INBTCompound INBTCompound}
268     */
269    public INBTCompound getCompound(String key);
270
271    /**
272     * Returns the direct child element of this tag which has
273     * the specified key as name as boolean, if possible.
274     *
275     * @param key the name of the child to retrieve
276     * @return the direct child element with the specified name as boolean
277     */
278    public boolean getBoolean(String key);
279
280    /**
281     * Returns the direct child element of this tag which has
282     * the specified key as name as an {@link INBTList INBTList}, if possible.
283     *
284     * @deprecated Magic value of type. Prefer using {@link #getList(String, TagType)}.
285     * @param key the name of the child to retrieve
286     * @param type the type of the elements of the list to retrieve
287     * @return the direct child element with the specified name as an {@link INBTList INBTList}
288     */
289    @Deprecated
290    public INBTList getList(String key, byte type);
291
292    /**
293     * Returns the direct child element of this tag which has
294     * the specified key as name as an {@link INBTList INBTList}, if possible.
295     *
296     * @param key the name of the child to retrieve
297     * @param type the type of the elements of the list to retrieve
298     * @return the direct child element with the specified name as an {@link INBTList INBTList}
299     */
300    public default INBTList getList(String key, TagType type) {
301        return getList(key, type.getId());
302    }
303
304    /**
305     * Removes any direct child element of this tag which has
306     * the specified key as name.
307     *
308     * @param key the name of the child to remove
309     */
310    public void remove(String key);
311
312    /**
313     * Removes all of the mappings from this tag.
314     * The tag will be empty after this call returns.
315     */
316    public void clear();
317}
